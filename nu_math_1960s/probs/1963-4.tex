\begin{problem}
    $x$の整式$x^n-1$を${(x-1)}^3$で割ったときの余りを求めよ。
\end{problem}

$x^n - 1 = (x-1)^3 Q_n(x) + ax^2 + bx + c$

$x=1$のとき\qquad \houteisiki{a + b + c = 0 \label{196341}}

両辺を$x$で微分して，
\[nx^{n-1} = \{3(x-1)^2Q_n(x) + (x-1)^3Q_n'(x)\} + 2ax + b\]

$x=1$のとき\qquad \houteisiki{n = 2a + b\label{196342}}

さらに両辺を$x$で微分して，
\begin{align*}
    n(n-1)x^{n-2} =
    \{6(x-1)Q_n(x) + 6(x-1)^2Q_n'(x) +(x-1)^3Q_n''(x) \}+ 2a
\end{align*}

$x=1$のとき\qquad \houteisiki{n(n-1) = 2a \label{196343}}

\eqref{196343}より，$a = \bunsuu{n(n-1)}{2}$

\eqref{196342}より，\begin{align*}
    n &= n(n-1) + b\\
    b &= -n^2 + 2n
\end{align*}

\eqref{196341}より，\begin{align*}
    \bunsuu{n(n-1)}{2} - n^2 + 2n + c &= 0\\
    n^2 - n -2n^2 + 4n + c &= 0 \\
    2c &= n^2 - 3n\\
    c &= \bunsuu{n^2 - 3n}{2}
\end{align*}
